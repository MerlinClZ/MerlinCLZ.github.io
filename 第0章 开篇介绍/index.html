01 链表的中间节点

876. 链表的中间结点

方法: 快慢指针 快指针两步, 慢指针一步

有两种写法;

1.链表长度为偶数时返回中间节点的前一个节点while(fast->next&&fast->next->next)

2.链表长度为偶数时返回中间节点的后一个节点while(fast&&fast->next)

class Solution &#123;
public:
   ListNode* middleNode(ListNode* head) &#123;
       ListNode* fast =head,*slow=head;
       while(fast&&fast->next)
       &#123;
           slow=slow->next;
           fast=fast->next->next;
       &#125;
       return slow;
   &#125;
&#125;;

04 快速排序

#include <iostream>
#include <vector>
using namespace std;
// 快速排序算法
void quickSort(vector<int>& arr, int low, int high) &#123;
    if (low < high) &#123;
        // 选取基准值
        int pivot = arr[high];
        int i = low - 1;
        
        // 分区过程
        for (int j = low; j <= high - 1; j++) &#123;
            if (arr[j] < pivot) &#123;
                i++;
                swap(arr[i], arr[j]);
            &#125;
        &#125;
        swap(arr[i + 1], arr[high]);
        int pivotIndex = i + 1;
        
        // 递归排序左右子数组
        quickSort(arr, low, pivotIndex - 1);
        quickSort(arr, pivotIndex + 1, high);
    &#125;

class Solution 
&#123;
public:
   bool isValid(string s) 
   &#123;
       stack<char>S;
       for (char ch : s)
       &#123;
           //如果当前栈为空，但是ch是右括号应该返回false
           if (S.empty() && ch == ')' || S.empty() && ch == ']'             || S.empty() && ch == '&#125;')
               return false;
           //如果ch是左括号，应该将其入栈，等待匹配
           if (ch == '(' || ch == '[' || ch == '&#123;')
               S.push(ch);
           
           else
           &#123;
               //如果输入的右括号和当前栈顶元素匹配，那么弹出栈顶元素
               if (S.top() == '(' && ch == ')' || S.top() ==                   '[' && ch == ']' || S.top() == '&#123;' && ch == '&#125;')
                   S.pop();
               //当前栈顶元素与ch不匹配，直接返回false
               else
                   return false;
           &#125;
       &#125;
       //如果栈为空，返回true
       if (S.empty())
           return true;
       else
           return false;
